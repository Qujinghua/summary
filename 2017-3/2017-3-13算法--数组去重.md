    function removeDuplicate( arr ) {
        var len = arr.length,
            temp = [];
    
        for ( var i = 0; i < len; i+=1 ) {
            if ( temp.indexOf(arr[i]) === -1 ) {
                temp.push(arr[i]);
            }
        }
        return temp;
    }

(indexOf() 方法可返回某个指定的字符串值在字符串中首次出现的位置。如果是第一次出现则返回-1)
